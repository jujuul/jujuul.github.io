这个图标切换和阴影背景的显示隐藏真是够麻烦的

const query = wx.createSelectorQuery()
query.select('#boxFixed').boundingClientRect()
query.selectViewport().scrollOffset()
query.exec((res) => {
    this.menutop = res[0].top
})

用来返回选中节点的TOP位置

wx.pageScrollTo({

    duration: 300
})
